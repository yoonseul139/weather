    <script>
      // 현재 날짜를 가져와 달력에 표시하는 JavaScript 코드입니다.
      // Java 코드의 'main' 함수 역할을 하며 동적인 기능을 제공합니다.

      function updateCalendar() {
        const now = new Date();

        // 1. 월 정보 설정
        const monthNames = [
          "JAN",
          "FEB",
          "MAR",
          "APR",
          "MAY",
          "JUN",
          "JUL",
          "AUG",
          "SEP",
          "OCT",
          "NOV",
          "DEC",
        ];
        const currentMonth = monthNames[now.getMonth()];

        // 2. 일 정보 설정 (10의 자리, 1의 자리)
        const currentDay = now.getDate();
        const dayString = currentDay.toString().padStart(2, "0"); // 1~9일은 앞에 0을 붙여 2자리로 만듭니다.
        const dayTens = dayString.charAt(0);
        const dayOnes = dayString.charAt(1);

        // 3. HTML 요소 업데이트
        document.querySelector("#month-card .card-text").textContent =
          currentMonth;
        document.querySelector("#day1-card .card-text").textContent = dayTens;
        document.querySelector("#day2-card .card-text").textContent = dayOnes;
      }

      // 페이지 로드 시 달력 업데이트 함수 실행
      document.addEventListener("DOMContentLoaded", updateCalendar);
